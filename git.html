Setup & Configuration

# Configure user information
git config --global user.name "Your Name"
git config --global user.email "your.email@example.com"

# View configuration
git config --list
git config user.name

# Initialize a repository
git init
git init my-project

# Clone a repository
git clone https://github.com/user/repo.git
git clone https://github.com/user/repo.git my-folder

Basic Commands

Working Directory

# Check status
git status
git status -s    # Short format

# Add files to staging
git add file.txt
git add .        # Add all files
git add *.js     # Add all JS files

# Commit changes
git commit -m "Commit message"
git commit -am "Add and commit"  # Add and commit tracked files

Viewing Changes

# View differences
git diff                    # Working directory vs staging
git diff --staged          # Staging vs last commit
git diff HEAD              # Working directory vs last commit
git diff branch1 branch2   # Between branches

# View file at specific commit
git show HEAD:file.txt
git show commit-hash:file.txt

Branching & Merging

# List branches
git branch              # Local branches
git branch -r           # Remote branches
git branch -a           # All branches

# Create and switch branches
git branch feature-x           # Create branch
git checkout feature-x         # Switch to branch
git checkout -b feature-x      # Create and switch
git switch feature-x           # Switch (newer command)
git switch -c feature-x        # Create and switch

# Merge branches
git checkout main
git merge feature-x            # Merge feature-x into main
git merge --no-ff feature-x    # Force merge commit

# Delete branches
git branch -d feature-x        # Delete merged branch
git branch -D feature-x        # Force delete branch

Remote Repositories

# Manage remotes
git remote -v                      # List remotes
git remote add origin url          # Add remote
git remote remove origin           # Remove remote
git remote rename origin upstream  # Rename remote

# Push and pull
git push origin main              # Push to remote
git push -u origin main           # Push and set upstream
git push --all                    # Push all branches
git pull origin main              # Pull from remote
git pull                          # Pull from upstream

# Fetch updates
git fetch origin                  # Fetch all branches
git fetch origin main             # Fetch specific branch

Viewing History

# View commit history
git log                           # Full log
git log --oneline                 # Compact log
git log --graph                   # Show branch graph
git log --all --graph --decorate  # Detailed graph

# Filter history
git log -n 5                      # Last 5 commits
git log --since="2 weeks ago"     # Time-based filter
git log --author="John"           # By author
git log --grep="fix"              # By commit message
git log file.txt                  # Changes to specific file

# Show specific commit
git show commit-hash              # Show commit details
git show HEAD~2                   # Show 2 commits ago

Undoing Changes

Working Directory

# Discard changes in working directory
git checkout file.txt             # Restore single file
git checkout .                    # Restore all files
git restore file.txt              # Newer command
git restore .                     # Restore all files

# Remove untracked files
git clean -f                      # Remove untracked files
git clean -fd                     # Remove files and directories

Staging Area

# Unstage files
git reset file.txt                # Unstage single file
git reset                         # Unstage all files
git restore --staged file.txt     # Newer command

Commits

# Amend last commit
git commit --amend -m "New message"

# Reset commits
git reset --soft HEAD~1           # Keep changes staged
git reset --mixed HEAD~1          # Keep changes unstaged
git reset --hard HEAD~1           # Discard changes

# Revert commits (creates new commit)
git revert commit-hash
git revert HEAD                   # Revert last commit

Stashing

# Stash changes
git stash                         # Stash tracked files
git stash -u                      # Include untracked files
git stash push -m "WIP: feature"  # Stash with message

# List and show stashes
git stash list                    # List all stashes
git stash show                    # Show latest stash
git stash show stash@{1}          # Show specific stash

# Apply stashes
git stash pop                     # Apply and remove latest
git stash apply                   # Apply without removing
git stash apply stash@{1}         # Apply specific stash

# Delete stashes
git stash drop                    # Delete latest stash
git stash drop stash@{1}          # Delete specific stash
git stash clear                   # Delete all stashes

Advanced Commands

Rebasing

# Rebase branch
git rebase main                   # Rebase current branch onto main
git rebase --interactive HEAD~3   # Interactive rebase last 3 commits
git rebase --abort               # Abort rebase
git rebase --continue            # Continue after resolving conflicts

Cherry-picking

# Cherry-pick commits
git cherry-pick commit-hash       # Apply specific commit
git cherry-pick commit1 commit2   # Apply multiple commits
git cherry-pick --no-commit hash  # Apply without committing

Tags

# Create tags
git tag v1.0                      # Lightweight tag
git tag -a v1.0 -m "Version 1.0"  # Annotated tag

# List and show tags
git tag                           # List all tags
git show v1.0                     # Show tag details

# Push tags
git push origin v1.0              # Push specific tag
git push origin --tags            # Push all tags
